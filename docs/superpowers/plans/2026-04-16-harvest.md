# HARVEST Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the HARVEST habit tracker SPA in `example/harvest/` as a complete, working example for ai-starter-kit.

**Architecture:** A React 19 SPA using Vite 6, vanilla-extract for type-safe zero-runtime CSS, and date-fns for date math. All state lives in a React Context that syncs to localStorage. The UI is a single-column week grid with a full-screen habit management overlay.

**Tech Stack:** React 19, Vite 6, TypeScript strict, vanilla-extract, date-fns

```ts
import {
  startOfWeek,
  addDays,
  format,
  isSameDay,
  isToday,
} from 'date-fns';

export const DATE_FORMAT = 'yyyy-MM-dd';

export function getWeekStart(date: Date): Date {
  return startOfWeek(date, { weekStartsOn: 1 });
}

export function getWeekDays(weekStart: Date): Date[] {
  return Array.from({ length: 7 }, (_, i) => addDays(weekStart, i));
}

export function formatDate(date: Date): string {
  return format(date, DATE_FORMAT);
}

export function formatWeekRange(weekStart: Date): string {
  const end = addDays(weekStart, 6);
  const sameMonth = weekStart.getMonth() === end.getMonth();
  const sameYear = weekStart.getFullYear() === end.getFullYear();

  const startStr = format(weekStart, 'M 月 d 日');
  const endStr = format(end, sameMonth ? 'd 日' : 'M 月 d 日');

  if (!sameYear) {
    return `${format(weekStart, 'yyyy 年 M 月 d 日')} – ${format(end, 'yyyy 年 M 月 d 日')}`;
  }

  return `${startStr} – ${endStr}`;
}

export { isSameDay, isToday };
```

- [ ] **Step 6: Write storage.ts**

```ts
import type { HarvestStore } from './types';

const STORAGE_KEY = 'harvest-data-v1';

export function loadStore(): HarvestStore | null {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return null;
    const parsed = JSON.parse(raw) as HarvestStore;
    if (parsed.version !== 1) return null;
    if (!Array.isArray(parsed.habits) || !Array.isArray(parsed.checkIns)) {
      return null;
    }
    return parsed;
  } catch {
    return null;
  }
}

let writeTimeout: number | undefined;

export function saveStore(store: HarvestStore): void {
  window.clearTimeout(writeTimeout);
  writeTimeout = window.setTimeout(() => {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(store));
    } catch {
      // silent fail
    }
  }, 50);
}
```

```tsx
import {
  createContext,
  useContext,
  useState,
  useCallback,
  useEffect,
  type ReactNode,
} from 'react';
import type { Habit, CheckIn, HarvestStore } from '../lib/types';
import { loadStore, saveStore } from '../lib/storage';
import { generateId } from '../lib/id';

interface HarvestContextValue {
  habits: Habit[];
  checkIns: CheckIn[];
  addHabit: (name: string) => void;
  renameHabit: (id: string, name: string) => void;
  deleteHabit: (id: string) => void;
  toggleCheckIn: (habitId: string, date: string) => void;
}

const HarvestContext = createContext<HarvestContextValue | null>(null);

export function HarvestProvider({ children }: { children: ReactNode }) {
  const [habits, setHabits] = useState<Habit[]>([]);
  const [checkIns, setCheckIns] = useState<CheckIn[]>([]);

  useEffect(() => {
    const store = loadStore();
    if (store) {
      setHabits(store.habits);
      setCheckIns(store.checkIns);
    }
  }, []);

  useEffect(() => {
    const store: HarvestStore = {
      version: 1,
      habits,
      checkIns,
    };
    saveStore(store);
  }, [habits, checkIns]);

  const addHabit = useCallback((name: string) => {
    const trimmed = name.trim();
    if (!trimmed) return;
    setHabits((prev) => {
      if (prev.length >= 7) return prev;
      return [
        ...prev,
        {
          id: generateId(),
          name: trimmed,
          createdAt: new Date().toISOString(),
        },
      ];
    });
  }, []);

  const renameHabit = useCallback((id: string, name: string) => {
    const trimmed = name.trim();
    if (!trimmed) return;
    setHabits((prev) =>
      prev.map((h) => (h.id === id ? { ...h, name: trimmed } : h))
    );
  }, []);

  const deleteHabit = useCallback((id: string) => {
    setHabits((prev) => prev.filter((h) => h.id !== id));
    setCheckIns((prev) => prev.filter((c) => c.habitId !== id));
  }, []);

  const toggleCheckIn = useCallback((habitId: string, date: string) => {
    setCheckIns((prev) => {
      const exists = prev.some(
        (c) => c.habitId === habitId && c.date === date
      );
      if (exists) {
        return prev.filter(
          (c) => !(c.habitId === habitId && c.date === date)
        );
      }
      return [...prev, { habitId, date }];
    });
  }, []);

  return (
    <HarvestContext.Provider
      value={{
        habits,
        checkIns,
        addHabit,
        renameHabit,
        deleteHabit,
        toggleCheckIn,
      }}
    >
      {children}
    </HarvestContext.Provider>
  );
}

export function useHarvest() {
  const ctx = useContext(HarvestContext);
  if (!ctx) throw new Error('useHarvest must be used within HarvestProvider');
  return ctx;
}
```

```tsx
import { useMemo } from 'react';
import type { Habit, CheckIn } from '../lib/types';
import { getWeekDays, formatDate, isToday } from '../lib/date';
import { HabitRow } from './HabitRow';
import * as styles from './HabitGrid.css';

const WEEKDAY_LABELS = ['周一', '周二', '周三', '周四', '周五', '周六', '周日'];

interface HabitGridProps {
  habits: Habit[];
  checkIns: CheckIn[];
  weekStart: Date;
  onToggle: (habitId: string, date: string) => void;
}

export function HabitGrid({ habits, checkIns, weekStart, onToggle }: HabitGridProps) {
  const weekDays = useMemo(() => getWeekDays(weekStart), [weekStart]);

  const rows = useMemo(() => {
    return habits.map((habit) => {
      const checks: Record<string, boolean> = {};
      checkIns
        .filter((c) => c.habitId === habit.id)
        .forEach((c) => {
          checks[c.date] = true;
        });
      return { habit, checks };
    });
  }, [habits, checkIns]);

  return (
    <div>
      <div className={styles.grid}>
        <div />
        {weekDays.map((day, i) => {
          const today = isToday(day);
          return (
            <div
              key={i}
              className={`${styles.headerCell} ${today ? styles.todayHeader : ''}`}
            >
              {WEEKDAY_LABELS[i]}
              <span className={styles.dayNumber}>{day.getDate()}</span>
            </div>
          );
        })}
      </div>
      {rows.map((row) => (
        <HabitRow
          key={row.habit.id}
          data={row}
          weekDays={weekDays}
          onToggle={onToggle}
        />
      ))}
    </div>
  );
}
```

```tsx
import { useState, useRef } from 'react';
import type { Habit } from '../lib/types';
import * as styles from './HabitManager.css';

interface HabitManagerProps {
  habits: Habit[];
  onAdd: (name: string) => void;
  onRename: (id: string, name: string) => void;
  onDelete: (id: string) => void;
  onClose: () => void;
}

export function HabitManager({
  habits,
  onAdd,
  onRename,
  onDelete,
  onClose,
}: HabitManagerProps) {
  const [newName, setNewName] = useState('');
  const inputRef = useRef<HTMLInputElement>(null);

  const handleAdd = () => {
    if (!newName.trim()) return;
    onAdd(newName.trim());
    setNewName('');
    inputRef.current?.focus();
  };

  const atLimit = habits.length >= 7;

  return (
    <div className={styles.overlay}>
      <div className={styles.header}>
        <h2 className={styles.title}>管理习惯</h2>
        <button type="button" className={styles.close} onClick={onClose}>
          完成
        </button>
      </div>

      <div className={styles.form}>
        <input
          ref={inputRef}
          type="text"
          className={styles.input}
          placeholder="例如：早起、阅读、运动"
          value={newName}
          onChange={(e) => setNewName(e.target.value)}
          onKeyDown={(e) => {
            if (e.key === 'Enter') handleAdd();
          }}
          disabled={atLimit}
        />
        <button
          type="button"
          className={styles.addButton}
          onClick={handleAdd}
          disabled={atLimit || !newName.trim()}
        >
          添加
        </button>
      </div>
      {atLimit && (
        <p className={styles.limit}>已达到上限 7 个习惯</p>
      )}

      <div className={styles.list}>
        {habits.map((habit) => (
          <div key={habit.id} className={styles.item}>
            <input
              type="text"
              className={styles.habitInput}
              defaultValue={habit.name}
              onBlur={(e) => onRename(habit.id, e.target.value)}
              onKeyDown={(e) => {
                if (e.key === 'Enter') {
                  (e.target as HTMLInputElement).blur();
                }
              }}
            />
            <button
              type="button"
              className={styles.deleteBtn}
              onClick={() => onDelete(habit.id)}
              aria-label="删除"
            >
              删除
            </button>
          </div>
        ))}
      </div>
    </div>
  );
}
```

```tsx
import { useState } from 'react';
import { addWeeks, subWeeks } from 'date-fns';
import { HarvestProvider, useHarvest } from './context/HarvestContext';
import { getWeekStart } from './lib/date';
import { WeekNav } from './components/WeekNav';
import { HabitGrid } from './components/HabitGrid';
import { HabitManager } from './components/HabitManager';
import { EmptyState } from './components/EmptyState';
import * as styles from './App.css';

function HarvestApp() {
  const [weekStart, setWeekStart] = useState(() => getWeekStart(new Date()));
  const [managing, setManaging] = useState(false);
  const { habits, checkIns, toggleCheckIn, addHabit, renameHabit, deleteHabit } =
    useHarvest();

  return (
    <div className={styles.page}>
      {managing ? (
        <HabitManager
          habits={habits}
          onAdd={addHabit}
          onRename={renameHabit}
          onDelete={deleteHabit}
          onClose={() => setManaging(false)}
        />
      ) : (
        <>
          <WeekNav
            weekStart={weekStart}
            onPrev={() => setWeekStart((d) => subWeeks(d, 1))}
            onNext={() => setWeekStart((d) => addWeeks(d, 1))}
            onToday={() => setWeekStart(getWeekStart(new Date()))}
          />
          {habits.length === 0 ? (
            <EmptyState />
          ) : (
            <HabitGrid
              habits={habits}
              checkIns={checkIns}
              weekStart={weekStart}
              onToggle={toggleCheckIn}
            />
          )}
          <button
            type="button"
            className={styles.manageLink}
            onClick={() => setManaging(true)}
          >
            管理习惯
          </button>
        </>
      )}
    </div>
  );
}

export default function App() {
  return (
    <HarvestProvider>
      <HarvestApp />
    </HarvestProvider>
  );
}
```
